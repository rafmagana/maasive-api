Build subquery

Converts queries to something understandable by MongoDB

For instance, if the API gets this:

 attribute.gt = 5

This functions returns:

 { attribute : { $gt : 5 } }
function build_sub_query( query, key_array, value, last_key ) { var current_key = key_array.shift(); if ( current_key == 'id' ) { current_key = '_id'; } if (current_key) { if ( current_key == "regexp" ) { var parts = /^\/(.*)\/([^\/]*)$/.exec( value ); query = parts ? RegExp(parts[1], parts[2]) : RegExp(value); } else if ( MONGO_KEYS.indexOf(current_key) > -1 ) { query['$'+current_key] = type_cast_value( value, last_key ); } else if ( !!key_array.length ) { query[current_key] = query[current_key] || {}; query[current_key] = build_sub_query( query[current_key], key_array, value, current_key ); } else if ( current_key == "in" ) { if ( value ) { query["$in"] = _.map(value.split(","), function(v) { return type_cast_value(v, last_key); }); } else { query["$in"] = []; } } else if ( current_key == 'eql' ) { query = type_cast_value(value, last_key); } else { query[current_key] = type_cast_value( value, current_key ); } } else { query[current_key] = type_cast_value( value, current_key ); } return query; }

function EndPoint( api_table, app ) { var self = this; this.api_table = api_table; this.app = app; this.table_name = this.api_table.app_id + "_" + this.api_table.name; this.connection_config = { hostname: this.app.mongo_host, port: this.app.mongo_port, username: this.app.identifier, password: this.app.secret, db_name: this.app.identifier }; this.table_name = this.api_table.name; this.db = new Db(this.connection_config.db_name, new Server(this.connection_config.hostname, this.connection_config.port, {}), {native_parser:false}); console.log( "DATABASE".green, this.db.databaseName, this.db.serverConfig.host ); this.qb = QueryBuilder(this); return this; };

Filter instance

EndPoint.prototype.filterInstance = function( obj ) { var self = this; _.each( obj, function( value, key ) { if ( INTERNAL.indexOf(key) > -1 || TIMESTAMPS.indexOf(key) > -1 || self.api_table.schema[key] == 'Date' ) { if ( value ) { obj[key] = value.valueOf(); } else { obj[key] = value; } } else if ( !self.api_table.schema.hasOwnProperty( key ) ) { delete obj[key]; } }); return obj; };

EndPoint.prototype.cleanInstance = function( instance ) { var self = this; /* This must be DRYed out */ _.each( instance, function( raw, key ) { var type = self.api_table.schema[key]; if ("_id" == key || "id" == key ) { instance['_id'] = CastDataType._id( raw ); } else if ( "_deleted" == key ) { instance["_deleted"] = CastDataType.Boolean( raw ); } else if ( TIMESTAMPS.indexOf(key) > -1) { instance[key] = CastDataType.Date( raw ); } else if (!type) { delete instance[key]; } else if (type == "Float") { instance[key] = CastDataType.Float( raw ); } else if (type == "String") { instance[key] = CastDataType.String( raw ); } else if (type == "Integer") { instance[key] = CastDataType.Integer( raw ); } else if (type == "Boolean") { instance[key] = CastDataType.Boolean( raw ); } else if (type == "Date") { instance[key] = CastDataType.Date( raw ); } else if (type == 'EncryptedString') { /* var shasum = crypto.createHash('sha1'); shasum.update(self.app.identifier + raw + self.app.secret); instance[key] = shasum.digest('hex');*/ instance[key] = CastDataType.EncryptedString( raw ); } else if (!raw) { delete instance[key]; } else { instance[key] = raw; } }); return instance; };
